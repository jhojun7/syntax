#if문 (조건문)
    주어진 조건의 참, 거짓 여부에 따라 프로그램의 흐름을 결정할 수 있는 구문을 뜻함
    불리언 데이터를 반환하거나 불리언 데이터로 해석할 수 있는 표현식을 의미로
    /문법
        if(조건){ // 조건이 true일때 실행코드 }
    /문법
        if(조건){ // 조건이 true일때 실행코드 } else { // 조건이 false일때 실행코드 }
    /예시
        let num = 5
        if(num == 3){console.log("TRUE!")}else{console.log("FALSE!")}
        * num의 값이 3과 같다면 TRUE! 를 그렇지 않다면 FALSE! 를 출력하라 // FALSE!
    
#while문 (반복문)
    비슷하거나 동일한 구문을 반복해서 수행할 수 있는 구문
    while문은 '주어진 조건이 참일 동안에 구문을 반복'하는 반복문이다.
    /문법
        while(조건){ // 조건이 true인 동안에 반복 수행할 코드 }
    조건이란 불리언을 반환하거나 불리언으로 해석될 수 있는 표현식
    반복 구문은 '루프'라고도 한다.
    /예시
        let num = 1
        while(num < 3){             * num가 3보다 작은 동안에는 반복해라
            console.log(num)        * num을 출력        
            num+=1 }                * num을 1씩 증가     // 1 2

#for문 (반복문)
    구문 작성시 반복을 위해 필요한 세가지 요소를 한 곳에 모아 작성함으로써 보다 명시적으로 반복 횟수를 표현할 수 있는 직관적인 구문
    /문법
        for(초기식; 조건식; 반복식){ // 조건이 true인 경우 반복 수행할 코드 }
        * 1초기식 2조건식 3수행코드 4반복식 1초기식 2조건식 ...
    초기식 : 반복 조건의 초기화 작업
    반복식 : 반복이 한 번 끝날 때마다 실행될 작업
    /예시
        for(let i = 1; i <= 3; i += 1) {        * i 는 1이고, 3보다 작거나 같다면 아래로 넘어가고 1씩 추가
            console.log(i) }                    * i 값 출력     // 1 2 3
